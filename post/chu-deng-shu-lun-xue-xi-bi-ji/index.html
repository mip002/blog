阶

若满足同余式 an≡1( mod m)a^n\equiv1(\bmod m)an≡1(modm) 的最小正整数存在，则这个 nnn 被称作 aaa 模 mmm 的阶，记作 δm(a)\delta_m(a)δm​(a)。

若 gcd⁡(a,n)=1\gcd(a,n)=1gcd(a,n)=1，且 ak≡1( mod n)a^k\equiv 1(\bmod n)ak≡1(modn)，则 k∣φ(n)k\mid \varphi(n)k∣φ(n)

定义

正整数 ggg 是正整数 nnn 的原根，当且仅当 1≤g≤n−11\leq g \leq n-11≤g≤n−1，且 ggg 模 nnn 的阶为 φ(n)\varphi(n)φ(n)。

什么数有原根

只有 2,4,pk,2×pk2,4,p^k,2\times p^k2,4,pk,2×pk（ppp 为奇素数）这类数才有原根。

先找到 nnn 的最小原根 ggg，那么 nnn 的每一个原根都形如 gkg^kgk 的形式，其中 kkk 满足 gcd⁡(k,φ(n))=1\gcd(k,\varphi(n))=1gcd(k,φ(n))=1，于是我们还可以知道其原根个数为 φ(φ(n))\varphi(\varphi(n))φ(φ(n))。

找最小原根

据说 nnn 的最小原根不超过 n0.25n^{0.25}n0.25 ，因此我们可以直接枚举 ggg，具体方法如下：

根据定义，若 ggg 是 nnn 的原根，需要满足两个条件：

  • gφ(n)≡1( mod n)g^{\varphi(n)}\equiv 1(\bmod n)gφ(n)≡1(modn)

	for(int i=2;i<=maxn;i++)
	{
		if(!vis[i]) prime[++tot]=i;
		for(int j=1;j<=tot;j++)
		{
			if(1ll*prime[j]*i>maxn) break;
			vis[prime[j]*i]=1;
			if(i%prime[j]==0) break; 
		}
	}
}

bool is[maxn+5];
inline void init() //预处理出有原根的数
{
	is[2]=is[4]=1;
	for(int i=2;i<=tot;i++)
	{
		ll x=1;
		for(int j=1;;j++)
		{
			x*=prime[i];
			if(x>maxn) break;
			is[x]=1;
			if(2*x<=maxn) is[2*x]=1;
		}
	}
}

inline ll qpow(ll x,ll y,ll mod)
{
	ll res=1;
	while(y)
	{
		if(y&1) res=res*x%mod;
		x=x*x%mod;
		y>>=1;
	}
	return res;
}

inline int phi(int x) 
{
	int res=x;
	for(int i=2;i*i<=x;i++)
	{
		if(x%i==0)
		{
			res=res/i*(i-1);
			while(x%i==0) x/=i;
		}
	}
	if(x>1) res=res/x*(x-1);
	return res;
}

int factor[maxn+5],num;
inline void divide(int x)
{
	num=0;
	for(int i=2;i*i<=x;i++)
	{
		if(x%i==0)
		{
			factor[++num]=i;
			if(i!=x/i) factor[++num]=x/i;
		}
	}
	if(x>1) factor[++num]=x;
}

inline bool check(int a,int phin,int mod)
{
	for(int i=1;i<=num;i++)
	{
		if(qpow(a,phin/factor[i],mod)==1) return false;
	}
	return true;
}

int t,n,d,ans[maxn+5],cnt;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0),cout.tie(0);
	pri();
	init();
	cin>>t;
	while(t--)
	{
		cnt=0;
		cin>>n>>d;
		if(!is[n])
		{
			cout<<"0\n\n";
			continue;
		}
		int phin=phi(n),g;
		divide(phin);
		for(int i=1;i<n;i++)
		{
			if(qpow(i,phin,n)==1&&check(i,phin,n))
			{
				g=i;
				break;
			}
		}
		int x=1;
		for(int i=1;i<=phin;i++)
		{
			x=1ll*x*g%n;
			if(__gcd(i,phin)==1) ans[++cnt]=x;
		}
		sort(ans+1,ans+1+cnt);
		cout<<cnt<<"\n";
		for(int i=d;i<=cnt;i+=d) cout<<ans[i]<<" ";
		cout<<"\n";
	}
	return 0;
}

	{
		if(n&1) res=res*a%mod;
		a=a*a%mod;
		n>>=1;
	}

ll p,b,n;

map<ll,ll> hmap;
inline ll bsgs()
{
	ll t=sqrt(p);
	for(int i=0;i<t;i++) hmap[n*qpow(b,i,p)%p]=i;
	b=qpow(b,t,p);
	if(b==0) return n==0?1:-1;
	for(int i=0;i<=t;i++) 
	{
		ll tmp=qpow(b,i,p);
		if(hmap[tmp]&&i*t-hmap[tmp]>=0) return i*t-hmap[tmp]; 
	}
	return -1;
}
